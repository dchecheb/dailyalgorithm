
# 선 긋기

```python
from sys import stdin

cnt = int(stdin.readline())  # 선 그은 총 회수
lines = list()               # 그어진 선 list
for _ in range(cnt):
    lines.append(sorted(map(int, stdin.readline().split())))
lines.sort()

minDot, maxDot = lines.pop(0)
length = maxDot - minDot

for start, end in lines:
    if maxDot <= start:
        length += end - start
        maxDot = end
        continue
    if end <= maxDot:
        continue
    else:
        length += end - maxDot
        maxDot = end
        
print(length)
```
